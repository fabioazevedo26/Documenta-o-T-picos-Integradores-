// --- FUNÇÃO AUXILIAR PARA ENVIO AO THINGSPEAK ---
void enviarAoThingSpeak() {
  // Define os campos do ThingSpeak
  ThingSpeak.setField(1, ppm_final);
  ThingSpeak.setField(2, temp_atual);
  ThingSpeak.setField(3, umidade_atual);

  // Envia os dados para a nuvem
  int codigoResposta = ThingSpeak.writeFields(myChannelNumber, myWriteAPIKey);
  
  if(codigoResposta == 200){
    Serial.println("[THINGSPEAK] Dados enviados com sucesso!");
  }
  else{
    Serial.println("[THINGSPEAK] Erro no envio. Código HTTP: " + String(codigoResposta));
  }
}

void setup() {
  Serial.begin(115200);

  pinMode(PINO_5V, OUTPUT);
  digitalWrite(PINO_5V, LOW); 

  pinMode(PINO_TESTE, INPUT_PULLUP); 

  analogSetAttenuation(ADC_0db); 

  WiFiManager wm;
  wm.setConfigPortalTimeout(180); 
  if(!wm.autoConnect("ESP32_Monitor_CO")) {
      Serial.println("Timeout na configuração. Reiniciando...");
      delay(3000);
      ESP.restart();
  }
  Serial.println("Conectado! IP: " + WiFi.localIP().toString());

  clientSSL.setInsecure();
  
  ThingSpeak.begin(clientTS); // <-- 3. Inicializa o ThingSpeak com o cliente padrão

  Wire.begin(); 

  if (!bme.begin(0x76)) {
      Serial.println("Erro: BME280 não detectado! Verifique fiação I2C.");
  } else {
      Serial.println("BME280 detectado com sucesso!");
  }

  server.on("/", handleRoot);
  server.begin();
  
  delay(2000); 
  tempoReferencia = millis();
  tempoBME = millis();
}

void loop() {
  server.handleClient(); 

  unsigned long tempoAtual = millis();

  // --- LEITURA CONSTANTE DO BME280 (A cada 2 segundos) ---
  if (tempoAtual - tempoBME >= 2000) {
    float t = bme.readTemperature();
    float h = bme.readHumidity();
    
    if (!isnan(t) && !isnan(h) && h > 0.0) {
      temp_atual = t;
      umidade_atual = h;
    } else {
      Wire.begin();
    }
    tempoBME = tempoAtual;
  }

  // --- CONTROLE DOS CICLOS DO MQ-7 ---
  if (fase == 1) {
    status_fase = "Aquecendo Sensor (5V)";
    digitalWrite(PINO_5V, HIGH); 
    
    if (tempoAtual - tempoReferencia >= 60000) {
      fase = 2;
      tempoReferencia = tempoAtual;
    }
  } 
  else if (fase == 2) {
    status_fase = "Estabilizando (1.4V)";
    digitalWrite(PINO_5V, LOW); 

    if (tempoAtual - tempoReferencia >= 85000 && tempoAtual - tempoReferencia < 86000) {
       long somaRaw = 0;
       for(int i=0; i<20; i++) {
         somaRaw += analogRead(MQ7_ANALOG);
         delay(10);
       }
       
       float rawMedia = somaRaw / 20.0;
       Serial.printf("\n[DEBUG] Bruto Lido no ADC (0-4095): %.1f\n", rawMedia);

       float fatorUmidade = 1.0 + (umidade_atual - 33.0) * 0.0047;
       if (fatorUmidade <= 0) fatorUmidade = 1.0; 
       
       ppm_final = (rawMedia / 4095.0) * 500 / fatorUmidade;
       if (ppm_final < 0) ppm_final = 0;

       Serial.printf("Leitura Concluída: PPM=%.2f | Umi=%.1f%%\n", ppm_final, umidade_atual);

       // <-- 4. ENVIO PARA O THINGSPEAK REALIZADO AQUI (A cada 90s junto com o cálculo)
       enviarAoThingSpeak(); 

       // --- ANÁLISE DE ALERTAS DO SENSOR REAL ---
       if (ppm_final >= LIMITE_MAX_PPM && !alertaEnviado) {
           dispararAlerta = true;
       } 
       else if (ppm_final < (LIMITE_MAX_PPM - 5.0) && alertaEnviado && digitalRead(PINO_TESTE) == HIGH) {
           dispararNormalizado = true;
       }
    }

    if (tempoAtual - tempoReferencia >= 90000) {
      fase = 1;
      tempoReferencia = tempoAtual;
    }
  }

  // --- VERIFICAÇÃO DO JUMPER DE TESTE ---
  if (digitalRead(PINO_TESTE) == LOW && !alertaEnviado) {
    status_fase = "[MODO DE TESTE ATIVO]";
    ppm_final = 999.9; 
    dispararAlerta = true;
    
    // Força envio imediato ao ThingSpeak no modo de teste se você desejar rastrear no gráfico
    static unsigned long ultimoEnvioTeste = 0;
    if (millis() - ultimoEnvioTeste > 15000) { 
      enviarAoThingSpeak();
      ultimoEnvioTeste = millis();
    }
  } 
  else if (digitalRead(PINO_TESTE) == HIGH && alertaEnviado && ppm_final < (LIMITE_MAX_PPM - 5.0)) {
    static unsigned long debounceBotao = 0;
    if (millis() - debounceBotao > 1000) { 
      dispararNormalizado = true;
      debounceBotao = millis();
    }
  }

  // --- PROCESSAMENTO ASSÍNCRONO DO TELEGRAM ---
  if (dispararAlerta) {
    dispararAlerta = false; 
    String mensajeAlerta = "⚠️ *ALERTA DE PERIGO (SIMULAÇÃO)!* ⚠️\n\n";
    mensajeAlerta += "Níveis perigosos de Monóxido de Carbono (CO) detectados!\n";
    mensajeAlerta += "*Concentração:* " + String(ppm_final, 1) + " PPM\n";
    mensajeAlerta += "*Umidade:* " + String(umidade_atual, 1) + "%\n";
    mensajeAlerta += "*Temperatura:* " + String(temp_atual, 1) + "°C\n\n";
    mensajeAlerta += "Por favor, ventile o ambiente imediatamente!";
    
    if(bot.sendMessage(CHAT_ID, mensajeAlerta, "Markdown")) {
      Serial.println("[TELEGRAM] Alerta enviado com sucesso!");
      alertaEnviado = true;
    } else {
      Serial.println("[TELEGRAM] Falha ao enviar alerta.");
    }
  }

  if (dispararNormalizado) {
    dispararNormalizado = false; 
    String mensagemNormalizado = "✅ *Nível de CO Normalizado (SIMULAÇÃO)* ✅\n\n";
    mensagemNormalizado += "O nível de Monóxido de Carbono baixou para níveis seguros.\n";
    mensagemNormalizado += "*Concentração Atual:* " + String(ppm_final, 1) + " PPM";
    
    if(bot.sendMessage(CHAT_ID, mensagemNormalizado, "Markdown")) {
      Serial.println("[TELEGRAM] Aviso de normalização enviado!");
      alertaEnviado = false;
    } else {
      Serial.println("[TELEGRAM] Falha ao enviar aviso de normalização.");
    }
  }
}
